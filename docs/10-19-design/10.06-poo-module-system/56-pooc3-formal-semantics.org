* 10.06.56.10 Core Mathematical Model

#+begin_src typst
$
& M = "modules",quad S = "scopes",quad Sigma = "states" \
  & sigma_M : M -> S,quad sigma_Sigma : Sigma -> S \
  & (S,<=) " is a partial or der" \
  & quad forall a. a <= a \
  & quad a <= b and b <= c => a <= c \
  & quad a <= b and b <= a => a = b \
  & D subset.eq M times M,quad D(m,d) = "module " m " depends on " d \
  & B subset.eq M times M,quad B(a,b) = "branch siblings" \
  & P_m : Sigma -> "Prop",quad Q_m : Sigma -> "Prop",quad
    G_m : Sigma -> "Prop" \
  & "DepsCompleted"(s,m) equiv
    forall d. D(m,d) => "Completed"(s,d) \
  & "ScopeAllowed"(s,m) equiv sigma_M(m) <= sigma_Sigma(s) \
  & "BranchExclusive"(s) equiv
    forall a,b. B(a,b) => not (G_a(s) and G_b(s)) \
  & "CanStart"(s,m) equiv
    "DepsCompleted"(s,m)
    and "ScopeAllowed"(s,m)
    and P_m(s)
    and Q_m(s)
    and G_m(s) \
  & "StepSound" equiv
    forall s,s',m. "Step"(s,s',m) => "CanStart"(s,m) \
  & "DependencySound" equiv
    "CanStart"(s,m) and D(m,d) => "Completed"(s,d) \
  & "ScopeTransitive" equiv
    "CanStart"(s,m) and sigma_Sigma(s) <= p
    => sigma_M(m) <= p \
  & "BranchSound" equiv
    B(a,b) and "CanStart"(s,a) and "CanStart"(s,b)
    => bot \
  & r : M -> NN,quad
    D(m,d) => r(d) < r(m) \
  & "RankedDeps"(D,r) =>
    not D(m,m)
    and
    not (D(a,b) and D(b,a)) \
  & "Trace"(s_0,s_n) =
    "Step"(s_0,s_1,m_0);dots.h;"Step"(s_(n-1),s_n,m_(n-1)) \
  & "Preserves"(I) equiv
    forall s,s',m. I(s) and "Step"(s,s',m) => I(s') \
  & "TracePreserves"(I) equiv
    I(s_0) and "Trace"(s_0,s_n) => I(s_n)
$
#+end_src

* 10.06.56.20 Lean Shape

#+begin_src lean
structure CanStart (spec : FlowSpec Module Scope State)
    (state : State) (module : Module) : Prop where
  depsCompleted :
    (dependency : Module) ->
    spec.dependsOn module dependency ->
    spec.completed state dependency
  scopeAllowed :
    spec.scopeOrder.le (spec.moduleScope module) (spec.activeScope state)
  policyHolds : spec.policyAllows state module
  preconditionHolds : spec.precondition state module
  guardHolds : spec.guard state module

theorem start_sound (step : Step spec state) :
  CanStart spec state step.module

theorem no_sibling_branch_start
    (exclusive : BranchExclusiveAt spec state)
    (sibling : spec.branchSibling left right)
    (leftStart : CanStart spec state left)
    (rightStart : CanStart spec state right) :
    False

structure DependencyRank (spec : FlowSpec Module Scope State) where
  rank : Module -> Nat
  decreases :
    spec.dependsOn module dependency ->
    rank dependency < rank module

theorem no_self_dependency_of_rank :
  DependencyRank spec -> ¬ spec.dependsOn module module

inductive Trace (spec : FlowSpec Module Scope State) :
    State -> State -> Prop

theorem trace_preserves :
  Preserves spec invariant ->
  Trace spec start final ->
  invariant start ->
  invariant final
#+end_src

* 10.06.56.40 Boundary

C3 linearization produces an ordered frontier.  It does not prove that a
frontier item may execute.  PooC3 strategy must prove =CanStart= for the
selected module.
